data = [
    {"category": "appliances", "price_range": "high", "portable": False},
    {"category": "furniture", "price_range": "medium", "portable": True},
    {"category": "electronics", "price_range": "high", "portable": True},

    {"category": "stationery", "price_range": "low", "portable": True},
    {"category": "appliances", "price_range": "medium", "portable": True}
]

# Filter the data for appliances
appliances = [item["name"] for item in data if item["category"] == "appliances"]
