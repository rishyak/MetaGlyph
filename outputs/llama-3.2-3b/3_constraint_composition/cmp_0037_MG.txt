SELECT 
  x.id, 
  x.category, 
  x.status, 
  CASE 
    WHEN x.category IN ('A') AND x.status = 'active' THEN 1
    WHEN x.category IN ('B') AND x.status = 'active' THEN 1
    WHEN x.category IN ('A') AND x.status = 'pending' THEN 1
    WHEN x.category IN ('B') AND x.status = 'pending' THEN 1
    ELSE 0
  END AS selected
FROM 
  item x
