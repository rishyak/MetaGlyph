    selected_ids = []
    for item in dataset:
        if item['Category'] in ['A', 'B'] or item['Status'] == 'active':
            selected_ids.append(item['ID'])
    return {
        "selected_ids": selected_ids,
        "count": len(selected_ids)
    }
