def process_dataset(dataset):
    # Step 1: Filter for value >= 50
    filtered_dataset = [item for item in dataset if item['Value'] >= 50]
    
    # Step 2: Filter for category == 'A'
    category_filtered = [item for item in filtered_dataset if item['Category'] == 'A']
    
    # Step 3: Extract ids and count
    selected_ids = [item['ID'] for item in category_filtered]
    count = len(selected_ids)
    
    # Output
    output = {
        "selected_ids": selected_ids,
        "count": count
    }
    
    return json.dumps(output, indent=2)
