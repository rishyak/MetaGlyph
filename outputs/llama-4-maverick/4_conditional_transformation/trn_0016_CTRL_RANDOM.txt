```python
def transform_records(records):
    transformed_records = []
    for record in records:
        if record['type'] == 'premium':
            record['amount'] *= 1.1
        if record['region'] == 'EU':
            record['amount'] *= 1.2
        if not record['active']:
            record['status'] = 'archived'
        transformed_records.append(record)
    return transformed_records
