def transform_records(records):
    transformed_records = []
    for record in records:
        new_record = record.copy()  # Create a copy to avoid modifying the original record
        
        # Apply transformations
        if new_record['type'] == 'premium':
            new_record['amount'] *= 1.1
        
        if new_record['region'] == 'EU':
            new_record['amount'] *= 1.2
        
        if not new_record['active']:
            new_record['status'] = 'archived'
        
        transformed_records.append(new_record)
    
    return transformed_records
